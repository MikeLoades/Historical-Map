// Initialize the map var map = L.map('map').setView([20, 0], 2); // Add background map L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { attribution: '© OpenStreetMap contributors' }).addTo(map); // Layer for borders var borderLayer = L.geoJson(null).addTo(map); // Load GeoJSON based on selected year function loadBorders(year) { let geojsonFile = `borders/${year}.geojson`; // âœ… Now this path is correct // Adjust path if needed fetch(geojsonFile) .then(response => response.json()) .then(data => { borderLayer.clearLayers(); // Remove old borders borderLayer.addData(data); // Load new borders }) .catch(err => console.log("No data for year", year)); } // Create a timeline slider var slider = document.getElementById("year-slider"); var yearDisplay = document.getElementById("year-display"); slider.addEventListener("input", function() { yearDisplay.textContent = this.value; loadBorders(this.value); }); // Initial load loadBorders(1800);
